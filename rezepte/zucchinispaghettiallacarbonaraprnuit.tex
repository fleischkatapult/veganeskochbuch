\begin{recipe}{Zucchini-Spaghetti alla Carbonara für 2 Personen. Mit Empfehlung von nuit.}
	\ingredient{100 g weißes Mandelmus} Mithilfe eines Schneebesens und 
	\ingredient{240 ml Wasser} vermengen. 
	\ingredient{$frac{1}{2}$ Bund Petersilie} warschen, trocken schütteln und die Blätter feinhacken.
	\ingredient{160 g Räuchertofu} in kleine Würfel schneiden. 
	\ingredient{1 Zwiebel} und 
	\ingredient{1 Knoblauchzehe} schälen und fein hacken.
	\ingredient{2 EL Olivenöl} in einer Pfanne erhitzen. Räuchertofu darin ca. 3 Minuten anbraten. Zwiebel- und Knoblauchwürfel zugeben und 3 MInuten mitbraten. Mandelmus-Mix, 
	\ingredient{abgeriebene Schale von $frac{1}{2}$ Bio-Zitrone} und 
	\ingredient{1 TL Zitronensaft} und Petersilie dazugeben, kräftig mit
	\ingredient{jodiertem Salz} und 
	\ingredient{schwarzem Pfeffer aus der Mühle} würzen. Carbonare kurz aufkochen und andicken lassen.
	\ingredient{5 Zucchini} waschen und mit einem Spiralschneider zu Spaghetti schneiden. Mit 
	\ingredient{2 EL Olivenöl} und etwas Salz mischen, vorsichtig unter die heiße Soße heben und 1 Minute darin erhitzen. In Pastatellern anrichten, mit etwas Olivenöl beträufen und mit schwarzen Pfeffer übermahlen.
	\item{Schwierigkeitsgrad} Stufe 1
	\preparationtime Etwa 25 Minuten.
	\washingup Normal.

\end{recipe}
